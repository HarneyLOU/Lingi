from com.diamond.iain.javagame import Game
from com.diamond.iain.javagame.utils import GameConstants
from java.lang.reflect import Field
from java.lang.reflect import Modifier
x = 700

player=Game.getPlayer()
f = player.getClass().getDeclaredField("y")
f.setAccessible(1)
f.set(player, x)

g = GameConstants.getDeclaredField("missileYPos")
print(GameConstants.missileYPos)
g.setAccessible(1)

modifiersField = Field.getDeclaredField("modifiers")
modifiersField.setAccessible(1)
modifiersField.setInt(g, g.getModifiers() & ~Modifier.FINAL)
g.setAccessible(1)
g.set(None, x)
