    public void run() {
        java.net.ServerSocket serverSocket = null;
        try {
            serverSocket = new java.net.ServerSocket(5000);
        } catch (java.io.IOException e) {
            e.printStackTrace();
        }
        java.net.Socket socket = null;
        try {

            socket = serverSocket.accept();
        } catch (java.io.IOException e) {
            e.printStackTrace();
        }
        javax.script.ScriptEngineManager scriptEngineManager = new javax.script.ScriptEngineManager();
        javax.script.ScriptEngine engine = scriptEngineManager.getEngineByName("python");
        java.lang.System.out.println("Jython is listening!");

        java.util.Scanner scanner = null ;
        try {
            scanner = new java.util.Scanner(new java.io.BufferedInputStream(socket.getInputStream()));
        } catch (Exception ex) {
            ex.printStackTrace();
        }
        while (scanner.hasNextLine()) {
                java.lang.String comingLine = scanner.nextLine();
                java.lang.System.out.println(comingLine);
                try {
                    engine.eval(comingLine);
                } catch (Exception e) {
                    java.lang.System.out.println("Huh?");
                }
            }
        }
